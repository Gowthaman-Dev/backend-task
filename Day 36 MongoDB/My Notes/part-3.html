🔹 GET BY ID

export const getById = async (req, res) => {
 try {
  const user = await userModel.findById(req.params.id);

  if (!user) {
   return res.status(404).json({ message: "User not found" });
  }

  res.status(200).json(user);

 } catch (error) {
  res.status(500).json({ message: error.message });
 }
};

🔹 UPDATE

export const updateUser = async (req, res) => {
 try {
  const updatedUser = await userModel.findByIdAndUpdate(
   req.params.id,
   req.body,
   { new: true, runValidators: true }
  );

  if (!updatedUser) {
   return res.status(404).json({ message: "User not found" });
  }

  res.status(200).json(updatedUser);

 } catch (error) {
  res.status(500).json({ message: error.message });
 }
};

Important:

new: true → updated document return ஆகும்

runValidators: true → schema validation apply ஆகும்

🔹 DELETE

export const deleteUser = async (req, res) => {
 try {
  const user = await userModel.findByIdAndDelete(req.params.id);

  if (!user) {

  }

  res.status(200).json({ message: "Deleted successfully" });

 } catch (error) {
  res.status(500).json({ message: error.message });
 }
};

2️⃣ Advanced Schema Validation

const userSchema = new mongoose.Schema({
 userName: {
  type: String,
  required: [true, "Name is required"],
  trim: true,
  minlength: 3
 },
 userEmail: {
  type: String,
  required: true,
  unique: true,
  lowercase: true
 },
 userAge: {
  type: Number,
  required: true,
  min: 18,
  max: 60
 },
 role: {
  type: String,
  enum: ["admin", "manager", "user"],
  default: "user"
 }
},{ timestamps: true })

🔥 Validation Explanation

Property Meaning
trim space remove
minlength minimum letters
lowercase auto lowercase
enum only allowed values
default default value

3️⃣ Professional Error Handling (Industry Level)

Instead of repeating try/catch everywhere:

Create middleware.

middleware/errorHandler.js

const errorHandler = (err, req, res, next) => {
 res.status(500).json({
  success: false,
  message: err.message
 });
};

export default errorHandler;

In server.js:

app.use(errorHandler)

👉 This is production style.
